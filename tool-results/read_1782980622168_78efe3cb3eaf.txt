   352→   352→/* ─── BibTeX helpers ─── */
   353→   353→
   354→   354→function generateBibtexKey(ref: Reference): string {
   355→   355→  // Extract first author surname
   356→   356→  const authorsStr = ref.authors || '';
   357→   357→  const firstAuthor = authorsStr.split(',')[0].split(' ').pop()?.replace(/[^a-zA-Z]/g, '').toLowerCase() || 'unknown';
   358→   358→  // Extract year
   359→   359→  const year = String(ref.year || '0000');
   360→   360→  // Extract first significant word from title
   361→   361→  const stopWords = new Set(['a', 'an', 'the', 'of', 'in', 'on', 'for', 'and', 'or', 'to', 'is', 'are', 'with', 'by', 'from', 'at', 'as', 'its', 'their', 'this', 'that', 'which', 'be', 'has', 'have', 'was', 'were', 'been', 'being', 'not', 'but', 'can', 'will', 'do', 'does', 'did', 'no', 'all', 'any', 'some', 'such', 'into', 'how', 'what', 'where', 'when', 'who', 'why']);
   362→   362→  const titleWords = ref.title.replace(/[^a-zA-Z\s]/g, '').split(/\s+/).filter(w => w.length > 2 && !stopWords.has(w.toLowerCase()));
   363→   363→  const titleWord = (titleWords[0] || 'untitled').toLowerCase();
   364→   364→  return `${firstAuthor}${year}${titleWord}`;
   365→   365→}
   366→   366→
   367→   367→function referenceToBibtex(ref: Reference): string {
   368→   368→  const key = generateBibtexKey(ref);
   369→   369→  const lines: string[] = `@article{${key},`.split('\n');
   370→   370→  lines.push(`  title = {${ref.title}},`);
   371→   371→  lines.push(`  author = {${ref.authors}},`);
   372→   372→  if (ref.journal) lines.push(`  journal = {${ref.journal}},`);
   373→   373→  lines.push(`  year = {${ref.year}},`);
   374→   374→  if (ref.volume) lines.push(`  volume = {${ref.volume}},`);
   375→   375→  if (ref.issue) lines.push(`  number = {${ref.issue}},`);
   376→   376→  if (ref.pages) lines.push(`  pages = {${ref.pages}},`);
   377→   377→  if (ref.doi) lines.push(`  doi = {${ref.doi}},`);
   378→   378→  lines.push('}');
   379→   379→  return lines.join('\n');
   380→   380→}
