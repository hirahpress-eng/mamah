   295вҶ’   295вҶ’function MarkdownPreview({ content, stageId, onGenerateVisual }: { content: string; stageId: StageId; onGenerateVisual: (placeholderId: string) => void }) {
   296вҶ’   296вҶ’  const { sectionStages } = useArticleStore();

   301вҶ’   301вҶ’  const parts: React.ReactNode[] = [];

   305вҶ’   305вҶ’

   326вҶ’   326вҶ’
