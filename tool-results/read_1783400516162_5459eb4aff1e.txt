    99тЖТ    99тЖТ/** Poll a job-based API until done or error */
   100тЖТ   100тЖТasync function pollJobApi(
   101тЖТ   101тЖТ  baseUrl: string,
   102тЖТ   102тЖТ  jobId: string,
   103тЖТ   103тЖТ  maxAttempts = 200,
   104тЖТ   104тЖТ  intervalMs = 3000,
   105тЖТ   105тЖТ): Promise<Record<string, unknown> | null> {
   106тЖТ   106тЖТ  for (let i = 0; i < maxAttempts; i++) {
   107тЖТ   107тЖТ    try {
   108тЖТ   108тЖТ      const res = await fetch(`${baseUrl}?jobId=${encodeURIComponent(jobId)}`);
   109тЖТ   109тЖТ      if (!res.ok) {
   110тЖТ   110тЖТ        await new Promise((r) => setTimeout(r, intervalMs));
   111тЖТ   111тЖТ        continue;
   112тЖТ   112тЖТ      }
   113тЖТ   113тЖТ      const data = await res.json();
   114тЖТ   114тЖТ      if (data.status === 'done') return data.result ?? data;
   115тЖТ   115тЖТ      if (data.status === 'error') return null;
   116тЖТ   116тЖТ      // still running
   117тЖТ   117тЖТ      await new Promise((r) => setTimeout(r, intervalMs));
   118тЖТ   118тЖТ    } catch {
   119тЖТ   119тЖТ      await new Promise((r) => setTimeout(r, intervalMs));
   120тЖТ   120тЖТ    }
   121тЖТ   121тЖТ  }
   122тЖТ   122тЖТ  return null;
   123тЖТ   123тЖТ}
