   145→function ReferencesSummary() {
   146→  const { references, prevStep } = useArticleStore();
   147→  const selectedRefs = references.filter((r) => r.isSelected);
   148→  const [isOpen, setIsOpen] = useState(false);
   149→  const yearVals = selectedRefs.map((r) => parseInt(String(r.year))).filter((y) => !isNaN(y));
   150→  const minYear = yearVals.length > 0 ? Math.min(...yearVals) : '—';
   151→  const maxYear = yearVals.length > 0 ? Math.max(...yearVals) : '—';
   152→  const journalCount = new Set(selectedRefs.map((r) => r.journal).filter(Boolean)).size;
   153→
   154→  return (
   155→    <section>
   156→      <div className="flex items-center justify-between mb-3">
   157→        <div className="flex items-center gap-3">
   158→          <div className="flex items-center justify-center w-9 h-9 rounded-lg bg-emerald-100 text-emerald-700 dark:bg-emerald-950/50 dark:text-emerald-400">
   159→            <Library className="w-5 h-5" />
   160→          </div>
   161→          <div>
   162→            <h3 className="text-base font-semibold text-foreground">Selected References</h3>
   163→            <p className="text-sm text-muted-foreground">Review your selected literature sources</p>
   164→          </div>
   165→        </div>
   166→        <Badge variant="secondary" className="bg-emerald-100 text-emerald-700 dark:bg-emerald-950/50 dark:text-emerald-400 font-semibold text-sm px-3 py-1">
   167→          {selectedRefs.length} reference{selectedRefs.length !== 1 ? 's' : ''}
   168→        </Badge>
   169→      </div>
   170→
   171→      <Card className="overflow-hidden">
   172→        <Collapsible open={isOpen} onOpenChange={setIsOpen}>
   173→          <div className="px-6 py-4">
   174→            <div className="flex items-center justify-between">
   175→              <div className="flex items-center gap-2 flex-wrap">
   176→                {selectedRefs.length > 0 ? (
   177→                  <>
   178→                    <span className="text-sm text-muted-foreground">
   179→                      Spanning <span className="font-medium text-foreground">{minYear}–{maxYear}</span>
   180→                    </span>
   181→                    <Separator orientation="vertical" className="h-4" />
   182→                    <span className="text-sm text-muted-foreground">
   183→                      {journalCount} journal{journalCount !== 1 ? 's' : ''}
   184→                    </span>
   185→                  </>
   186→                ) : (
   187→                  <span className="text-sm text-amber-600 dark:text-amber-400">No references selected yet</span>
   188→                )}
   189→              </div>
   190→              <div className="flex items-center gap-2">
   191→                <Button variant="ghost" size="sm" onClick={() => prevStep()} className="text-emerald-700 hover:text-emerald-800 hover:bg-emerald-50 dark:text-emerald-400 dark:hover:bg-emerald-950/30">
   192→                  <Pencil className="w-3.5 h-3.5" /> Edit References
   193→                </Button>
   194→                <CollapsibleTrigger asChild>
   195→                  <Button variant="ghost" size="sm" className="text-muted-foreground">
   196→                    {isOpen ? <ChevronUp className="w-4 h-4" /> : <ChevronDown className="w-4 h-4" />}
   197→                  </Button>
   198→                </CollapsibleTrigger>
   199→              </div>
   200→            </div>
   201→          </div>
   202→          <AnimatePresence>

   204→              <CollapsibleContent forceMount>

   450→// ─── Markdown Preview with Visual Placeholders ──────────────────────────────────
   451→
   452→function MarkdownPreview({ content, stageId, onGenerateVisual }: { content: string; stageId: StageId; onGenerateVisual: (placeholderId: string) => void }) {
   453→  const { sectionStages } = useArticleStore();
   454→  const stage = sectionStages.find((s) => s.id === stageId);
   455→  const placeholders = stage?.visualPlaceholders || [];
   456→
   457→  // Split content by placeholder pattern and interleave with cards
   458→  const parts: React.ReactNode[] = [];
   459→  let lastIndex = 0;
   460→  const regex = new RegExp(PLACEHOLDER_REGEX.source, 'gi');
   461→  let match;
   462→
   463→  while ((match = regex.exec(content)) !== null) {
   464→    // Text before placeholder
   465→    if (match.index > lastIndex) {
   466→      parts.push(<div key={`md-${lastIndex}`} className="prose prose-sm dark:prose-invert max-w-none"><ReactMarkdown>{content.slice(lastIndex, match.index)}</ReactMarkdown></div>);
   467→    }
   468→    // Find matching placeholder
   469→    const phType = match[1].toLowerCase() as 'figure' | 'table';
   470→    const phDesc = match[2].trim();
   471→    const ph = placeholders.find((p) => p.description === phDesc && p.type === phType);
   472→    if (ph) {
   473→      parts.push(<VisualPlaceholderCard key={ph.id} placeholder={ph} onGenerate={onGenerateVisual} stageId={stageId} />);
   474→    } else {
   475→      parts.push(
   476→        <div key={`placeholder-${match.index}`} className="my-3 p-3 rounded-lg border border-dashed border-muted-foreground/30 text-sm text-muted-foreground">
   477→          <span className="capitalize">{phType}:</span> {phDesc}
   478→        </div>
   479→      );
   480→    }
   481→    lastIndex = match.index + match[0].length;
   482→  }
   483→
   484→  // Remaining text
   485→  if (lastIndex < content.length) {
   486→    parts.push(<div key={`md-end`} className="prose prose-sm dark:prose-invert max-w-none"><ReactMarkdown>{content.slice(lastIndex)}</ReactMarkdown></div>);
   487→  }
   488→
   489→  return <div className="space-y-1">{parts}</div>;
   490→}

   764→
   765→    updateStage(currentStage.id, {
   766→      status: 'generating',
   767→      error: undefined,

   769→      wordCount: 0,

   773→    setStatusText(`Preparing to generate ${currentStage.label}...`);

   776→      // 1. POST to create a generation job (returns immediately)

   778→        method: 'POST',
   779→        headers: { 'Content-Type': 'application/json' },
   780→        body: JSON.stringify({
   781→          stageId: currentStage.id,
   782→          title: effectiveTitle,
   783→          keywords: effectiveKeywords,
   784→          references: selectedRefs,
   785→          researchMethod,
   786→          additionalInstructions,
   787→          engineId: selectedEngine,
   788→          previousSections: completedStages.map((s) => ({ id: s.id, label: s.label, content: s.content })),
   789→        }),
   790→        signal: abort.signal,
   791→      });
   792→
   793→      const postData = await postRes.json();
   794→      if (!postData.success || !postData.jobId) {
   795→        throw new Error(postData.error || 'Failed to start generation job');
   796→      }
   797→
   798→      const jobId = postData.jobId;
   799→      setStatusText(`Generation job started, waiting for ${currentStage.label}...`);
   800→
   801→      // 2. Poll for result every 3 seconds
   802→      const POLL_INTERVAL = 3000;
   803→      const MAX_POLL_TIME = 10 * 60 * 1000; // 10 minutes max
   804→      const startTime = Date.now();
   805→
   806→      while (Date.now() - startTime < MAX_POLL_TIME) {
   807→        if (abort.signal.aborted) throw new DOMException('Aborted', 'AbortError');
   808→        await new Promise((r) => setTimeout(r, POLL_INTERVAL));
   809→        if (abort.signal.aborted) throw new DOMException('Aborted', 'AbortError');
   810→
   811→        const pollRes = await fetch(`/api/article/generate-section?jobId=${jobId}`, {
   812→          signal: abort.signal,
   813→        });
   814→        const pollData = await pollRes.json();
   815→
   816→        if (!pollData.success) throw new Error(pollData.error || 'Polling failed');
   817→
   818→        // Update status text from server
   819→        if (pollData.statusMessage) {
   820→          setStatusText(pollData.statusMessage);
   821→        }
   822→
   823→        if (pollData.status === 'done' && pollData.result) {
   824→          updateStage(currentStage.id, {
   825→            status: 'done',
   826→            content: pollData.result.content,
   827→            wordCount: pollData.result.wordCount || 0,
   828→            visualPlaceholders: (pollData.result.visualPlaceholders || []).map((p: { id: string; type: 'figure' | 'table'; description: string }) => ({
   829→              ...p,
   830→              generated: false,
   831→            })),
   832→            completedAt: new Date().toISOString(),
   833→          });
   834→          toast.success(`${currentStage.label} generated successfully!`);
   835→          return;
   836→        }
   837→
   838→        if (pollData.status === 'error') {
   839→          throw new Error(pollData.error || 'Generation failed on server');
   840→        }
   841→      }
   842→
   843→      throw new Error('Generation timed out after 10 minutes');
   844→    } catch (err: unknown) {
   845→      if (err instanceof DOMException && err.name === 'AbortError') {
   846→        updateStage(currentStage.id, { status: 'pending', error: undefined });
   847→        toast.info(`${currentStage.label} generation cancelled.`);
   848→        return;
   849→      }
   850→      const message = err instanceof Error ? err.message : 'Unknown error';
   851→      updateStage(currentStage.id, { status: 'error', error: message });
   852→      toast.error(`Failed to generate ${currentStage.label}: ${message}`);
   853→    } finally {
   854→      abortRef.current = null;
   855→    }

   897→      // POST to start visual generation job
   898→      const postRes = await fetch('/api/article/generate-visual', {
   899→        method: 'POST',
   900→        headers: { 'Content-Type': 'application/json' },
   901→        body: JSON.stringify({
   902→          type: placeholder.type,
   903→          description: placeholder.description,
   904→          context: stage.content.substring(0, 500),
   905→          articleTitle: effectiveTitle,
   906→          engineId: selectedEngine,
   907→        }),
   908→        signal: abort.signal,
   909→      });
   910→
   911→      const postData = await postRes.json();
   912→      if (!postData.success || !postData.jobId) {
   913→        failPlaceholder(postData.error || 'Failed to start visual generation');
   914→        toast.error(postData.error || 'Failed to start visual generation');
   915→        return;
   916→      }
   917→
   918→      const jobId = postData.jobId;
   919→
   920→      // Poll for result every 3 seconds
   921→      const POLL_INTERVAL = 3000;
   922→      const MAX_POLL_TIME = 5 * 60 * 1000;
   923→      const startTime = Date.now();
   924→
   925→      while (Date.now() - startTime < MAX_POLL_TIME) {
   926→        if (abort.signal.aborted) return;
   927→        await new Promise((r) => setTimeout(r, POLL_INTERVAL));
   928→        if (abort.signal.aborted) return;
   929→
   930→        const pollRes = await fetch(`/api/article/generate-visual?jobId=${jobId}`, {
   931→          signal: abort.signal,
   932→        });
   933→        const pollData = await pollRes.json();
   934→
   935→        if (!pollData.success) {
   936→          failPlaceholder(pollData.error || 'Polling failed');
   937→          toast.error(pollData.error || 'Polling failed');
   938→          return;
   939→        }
   940→
   941→        if (pollData.status === 'done' && pollData.result) {
   942→          const updatedPlaceholders = stage.visualPlaceholders.map((p) =>
   943→            p.id === placeholderId ? { ...p, generated: true, data: pollData.result.data, error: undefined } : p
   944→          );
   945→          updateStage(stage.id, { visualPlaceholders: updatedPlaceholders });
   946→          toast.success(`${placeholder.type === 'figure' ? 'Figure' : 'Table'} generated!`);
   947→          return;
   948→        }
   949→
   950→        if (pollData.status === 'error') {
   951→          failPlaceholder(pollData.error || 'Generation failed');
   952→          toast.error(pollData.error || 'Generation failed');
   953→          return;
   954→        }
   955→      }
   956→
   957→      failPlaceholder('Visual generation timed out');
   958→      toast.error('Visual generation timed out');
   959→    } catch (err: unknown) {
   960→      if (err instanceof DOMException && err.name === 'AbortError') return;
   961→      const message = err instanceof Error ? err.message : 'Unknown error';
   962→      failPlaceholder(message);
   963→      toast.error(message);
   964→    }
   965→  }, [sectionStages, currentStage, effectiveTitle, selectedEngine, updateStage]);
   966→
   967→  const handleCompile = useCallback(() => {
   968→    compileArticleFromStages();
   969→    nextStep();
   970→    toast.success('Article compiled! Moving to preview.');
   971→  }, [compileArticleFromStages, nextStep]);
   972→
   973→  const handleRegenerateAll = useCallback(() => {
   974→    resetStages();
   975→    setCurrentStageIndex(0);
   976→    setView('generating');
   977→  }, [resetStages, setCurrentStageIndex]);
   978→
   979→  // ─── View 1: Configuration ──────────────────────────────────────────────────
   980→  if (view === 'config') {
   981→    return (

   984→        <div className="flex items-center justify-center">
   985→          <div className="flex items-center gap-0">
   986→            <div className="flex items-center justify-center w-8 h-8 rounded-full bg-emerald-500 text-white"><Check className="size-3.5" /></div>
   987→            <div className="w-12 h-0.5 bg-emerald-500" />
   988→            <div className="flex items-center justify-center w-8 h-8 rounded-full bg-emerald-500 text-white"><Check className="size-3.5" /></div>
   989→            <div className="w-12 h-0.5 bg-emerald-500" />
   990→            <div className="flex items-center justify-center w-8 h-8 rounded-full bg-emerald-500 text-white pulse-dot"><span className="text-[10px] font-bold">3</span></div>
   991→          </div>
   992→        </div>
   993→        <div className="flex items-center justify-center gap-8 text-[10px] font-medium text-muted-foreground -mt-1">
   994→          <span>Define</span><span>References</span><span className="text-emerald-600 dark:text-emerald-400">Generate</span>
   995→        </div>
   996→
   997→        <ReferencesSummary />
   998→        <Separator />
   999→        <MethodSelection />
  1000→        <AnimatePresence>
  1001→          {researchMethod && (
  1002→            <div className="mt-4">
  1003→              <MethodologyWorkflowPanel
  1004→                methodId={researchMethod}
  1005→                methodName={RESEARCH_METHODS.find((m) => m.id === researchMethod)?.title ?? researchMethod.replace(/-/g, ' ')}
  1006→              />
  1007→            </div>
  1008→          )}
  1009→        </AnimatePresence>
  1010→        <Separator />
  1011→
  1012→        {/* Engine Selector */}
  1013→        <section>
  1014→          <div className="flex items-center gap-3 mb-3">
  1015→            <div className="flex items-center justify-center w-9 h-9 rounded-lg bg-emerald-100 text-emerald-700 dark:bg-emerald-950/50 dark:text-emerald-400"><Sparkles className="w-5 h-5" /></div>
  1016→            <div>

  1051→          <div className="flex items-center gap-3 mb-3">
  1052→            <div className="flex items-center justify-center w-9 h-9 rounded-lg bg-emerald-100 text-emerald-700 dark:bg-emerald-950/50 dark:text-emerald-400"><Settings className="w-5 h-5" /></div>

  1071→                <div><span className="text-muted-foreground">References</span><p className="font-medium text-foreground mt-0.5">{selectedRefs.length} selected</p></div>

  1073→              </div>

  1081→                  <Clock className="w-4 h-4 text-emerald-600 dark:text-emerald-400 shrink-0" />

  1088→              </Button>
