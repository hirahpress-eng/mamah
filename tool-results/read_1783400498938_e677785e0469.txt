     1→/**
     2→ * Writing Flows — Complete cicil (incremental) generation configurations
     3→ *
     4→ * Defines the structure for each writing mode:
     5→ * - Skripsi (S1): BAB 1-5 + Referensi, 5-7 sub-steps per BAB
     6→ * - Tesis (S2): BAB 1-5 + Referensi, 5-8 sub-steps per BAB
     7→ * - Disertasi (S3): BAB 1-5 + Referensi, 8-11 sub-steps per BAB
     8→ * - Buku Ilmiah Indonesia: Book chapters
     9→ * - Buku Ilmiah English: Book chapters (English)
    10→ * - Buku Bahasa Arab: Arabic book structure
    11→ * - Buku Eksakta/Matematika: Science/Math book structure
    12→ * - Buku Keislaman: Islamic Studies book structure
    13→ * - Proposal Penelitian: Research proposal stages
    14→ * - Esai Beasiswa: Scholarship essay stages
    15→ * - Makalah: Paper stages
    16→ */
    17→
    18→// ─── Types ──────────────────────────────────────────────────────────────────
    19→
    20→export type CicilWritingMode =
    21→  | 'skripsi'
    22→  | 'tesis'
    23→  | 'disertasi'
    24→  | 'buku-id'
    25→  | 'buku-en'
    26→  | 'buku-arab'
    27→  | 'buku-eksakta'
    28→  | 'buku-keislaman'
    29→  | 'proposal'
    30→  | 'scholarship'
    31→  | 'paper';
    32→
    33→export interface CicilSubStep {
    34→  id: string;
    35→  label: string;
    36→  labelId: string;
    37→  targetWords: number;
    38→  promptFocus: string;
    39→}
    40→
    41→export interface CicilChapter {
    42→  id: string;
    43→  label: string;
    44→  labelId: string;
    45→  subSteps: CicilSubStep[];
    46→}

  1451→/** Get total number of sub-steps across all chapters */
  1452→export function getTotalSteps(mode: CicilWritingMode): number {
  1453→  return WRITING_FLOWS[mode].chapters.reduce(
  1454→    (sum, ch) => sum + ch.subSteps.length,
  1455→    0,
  1456→  );
  1457→}

  1459→/** Get flat list of all sub-steps with chapter info */
  1460→export function getFlatSteps(mode: CicilWritingMode): Array<{
  1461→  chapterId: string;
  1462→  chapterLabel: string;
  1463→  step: CicilSubStep;
  1464→  index: number;
  1465→}> {
  1466→  const steps: Array<{
  1467→    chapterId: string;
  1468→    chapterLabel: string;
  1469→    step: CicilSubStep;
  1470→    index: number;
  1471→  }> = [];
  1472→  let idx = 0;
  1473→  for (const ch of WRITING_FLOWS[mode].chapters) {
  1474→    for (const ss of ch.subSteps) {
  1475→      steps.push({ chapterId: ch.id, chapterLabel: ch.label, step: ss, index: idx });
  1476→      idx++;
  1477→    }
  1478→  }
  1479→  return steps;
  1480→}

  1482→/** Get total target word count */

  1486→    0,
