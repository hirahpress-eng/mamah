    27→// ─── Per-Section Retry with Exponential Backoff ────────────────────────────────
    28→
    29→const FALLBACK_PATTERN = /all ai engines are currently unavailable/i;
    30→const RETRY_DELAYS = [10000, 25000, 50000, 80000]; // ms — aggressive backoff to survive rate limits
    31→const MAX_SECTION_RETRIES = 5;
    32→
    33→// ─── Word Count Targets (18-page article = ~16,000 words body) ─────────────────
    34→// These are the MINIMUM word counts each section must achieve.
    35→// The AI will be asked to expand if below these thresholds.
    36→
    37→const SECTION_MIN_WORDS: Record<string, number> = {
    38→  abstract: 250,
    39→  introduction: 1500,
    40→  literature_review: 2500,
    41→  method: 1500,
    42→  results: 3000,
    43→  discussion: 2500,
    44→  conclusion: 600,
    45→};
    46→
    47→// The ideal TARGET word counts the user wants to see
    48→const SECTION_TARGET_WORDS: Record<string, number> = {
    49→  abstract: 300,
    50→  introduction: 2000,
    51→  literature_review: 3250,
    52→  method: 2000,
    53→  results: 4000,
    54→  discussion: 3250,
    55→  conclusion: 800,
    56→};
    57→
    58→function countWords(text: string): number {
    59→  return text.split(/\s+/).filter((w) => w.length > 0).length;
    60→}
    61→
    62→function isSectionValid(content: string): boolean {
    63→  if (!content || content.trim().length === 0) return false;
    64→  if (FALLBACK_PATTERN.test(content)) return false;
    65→  const words = countWords(content);
    66→  if (words < 100) return false;
    67→  if (words > 25000) return false; // catch garbage output
    68→  return true;
    69→}

    71→async function generateSectionWithRetry(
    72→  sectionName: string,
    73→  engine: AIEngineId,
    74→  systemPrompt: string,
    75→  userPrompt: string,
    76→  options?: { maxTokens?: number },
    77→): Promise<string> {
    78→  for (let attempt = 1; attempt <= MAX_SECTION_RETRIES; attempt++) {
    79→    const result = (await generateWithEngine(engine, systemPrompt, userPrompt, options)) || '';
    80→
    81→    if (isSectionValid(result)) {
    82→      console.log(
    83→        `[article-gen] ${sectionName}: ${result.length} chars, ${countWords(result)} words`,
    84→      );
    85→      return result;
    86→    }
    87→
    88→    // Log word count on failure for debugging
    89→    const wc = countWords(result);
    90→    console.warn(
    91→      `[article-gen] ${sectionName} attempt ${attempt}/${MAX_SECTION_RETRIES} invalid: ${result.length} chars, ${wc} words, empty=${!result.trim()}`,
    92→    );
    93→
    94→    // Content is empty/short/fallback — retry if attempts remain
    95→    if (attempt < MAX_SECTION_RETRIES) {
    96→      const delay = RETRY_DELAYS[attempt - 1];
    97→      console.log(
    98→        `[article-gen] Retrying ${sectionName} (attempt ${attempt + 1}/${MAX_SECTION_RETRIES})...`,
    99→      );
   100→      await new Promise<void>((r) => setTimeout(r, delay));
   101→    } else {
   102→      console.error(
   103→        `[article-gen] ${sectionName} failed after ${MAX_SECTION_RETRIES} attempts (final: ${wc} words)`,
   104→      );
   105→    }
   106→  }
   107→
   108→  return '';
   109→}

   111→// ─── Post-Generation Word Count Expansion ─────────────────────────────────────
   112→// If a section is below its target word count, this function sends an EXPANSION
   113→// prompt asking the AI to lengthen the section while maintaining quality.
   114→
   115→async function expandSectionToTarget(
   116→  sectionName: string,
   117→  sectionType: string,
   118→  currentContent: string,
   119→  engine: AIEngineId,
   120→  systemPrompt: string,
   121→  maxExpansionAttempts: number = 2,
   122→): Promise<string> {
   123→  const currentWords = countWords(currentContent);
   124→  const minWords = SECTION_MIN_WORDS[sectionType] || 500;
   125→  const targetWords = SECTION_TARGET_WORDS[sectionType] || minWords;
   126→
   127→  // If already at or above target, no expansion needed
   128→  if (currentWords >= targetWords) {
   129→    console.log(`[expand] ${sectionName}: ${currentWords}/${targetWords} words — NO expansion needed`);
   130→    return currentContent;
   131→  }
   132→
   133→  // If below minimum, definitely expand. If between min and target, also expand.
   134→  const deficit = targetWords - currentWords;
   135→  console.log(`[expand] ${sectionName}: ${currentWords}/${targetWords} words — EXPANDING by ~${deficit} words`);
   136→
   137→  let content = currentContent;
   138→
   139→  for (let attempt = 1; attempt <= maxExpansionAttempts; attempt++) {
   140→    const wordsNow = countWords(content);
   141→    const remainingDeficit = targetWords - wordsNow;
   142→
   143→    if (remainingDeficit <= 0) {
   144→      console.log(`[expand] ${sectionName}: reached ${wordsNow} words after expansion — SUCCESS`);
   145→      break;
   146→    }
   147→
   148→    const expansionPrompt = `## SECTION EXPANSION TASK
   149→
   150→The following ${sectionName} section currently has ${wordsNow} words but MUST reach at least ${targetWords} words (currently short by ~${remainingDeficit} words).
   151→
   152→YOUR TASK: Expand this section by adding approximately ${remainingDeficit} MORE words of high-quality academic content. Follow these rules precisely:
   153→
   154→1. **DO NOT delete or significantly rewrite** any existing content — ADD new material
   155→2. **Insert new paragraphs** in logical positions where they naturally fit the argument flow
   156→3. **Add more detailed analysis** — expand claims with additional evidence, deeper reasoning, and more nuanced discussion
   157→4. **Add more scholarly dialogue** — introduce additional citation synthesis where the argument can be strengthened
   158→5. **Expand existing paragraphs** that are thin (under 100 words) by adding supporting evidence and elaboration
   159→6. **Maintain the same writing quality** — formal British English, hedging language, sophisticated transitions
   160→7. **Do NOT add new section headings** — only expand within the existing structure
   161→8. **Do NOT include any preamble** — output the COMPLETE expanded section from start to finish
   162→
   163→CRITICAL: The output must be the COMPLETE ${sectionName} section (not just the new parts). It must read as a cohesive, naturally-flowing academic text.
   164→
   165→## CURRENT ${sectionName.toUpperCase()} SECTION (${wordsNow} words):
   166→
   167→${content}
   168→
   169→OUTPUT THE COMPLETE EXPANDED ${sectionName.toUpperCase()} SECTION ABOVE. No preamble. No commentary. Start from the very first sentence.`;
   170→
   171→    try {
   172→      const expandedResult = await generateWithEngine(
   173→        engine,
   174→        systemPrompt,
   175→        expansionPrompt,
   176→        { maxTokens: 32000 },
   177→      );
   178→
   179→      if (expandedResult && expandedResult.trim().length > 0 && !FALLBACK_PATTERN.test(expandedResult)) {
   180→        const newWordCount = countWords(expandedResult);
   181→        if (newWordCount > wordsNow) {
   182→          const improvement = newWordCount - wordsNow;
   183→          console.log(`[expand] ${sectionName} attempt ${attempt}: ${wordsNow} → ${newWordCount} words (+${improvement})`);
   184→          content = expandedResult;
   185→
   186→          if (newWordCount >= targetWords) {
   187→            console.log(`[expand] ${sectionName}: TARGET REACHED at ${newWordCount} words`);
   188→            break;
   189→          }
   190→        } else {
   191→          console.warn(`[expand] ${sectionName} attempt ${attempt}: no improvement (${newWordCount} words)`);
   192→        }
   193→      } else {
   194→        console.warn(`[expand] ${sectionName} attempt ${attempt}: invalid/empty result`);
   195→      }
   196→    } catch (err) {
   197→      console.error(`[expand] ${sectionName} attempt ${attempt} error:`, err);
   198→    }
   199→
   200→    // Wait between expansion attempts
   201→    if (attempt < maxExpansionAttempts) {
   202→      await new Promise<void>((r) => setTimeout(r, 10000));
   203→    }
   204→  }
   205→
   206→  const finalWords = countWords(content);
   207→  if (finalWords < minWords) {
   208→    console.warn(`[expand] ${sectionName}: WARNING — final ${finalWords} words is still below minimum ${minWords}`);
   209→  } else if (finalWords < targetWords) {
   210→    console.log(`[expand] ${sectionName}: final ${finalWords} words (above minimum ${minWords}, below target ${targetWords})`);
   211→  } else {
   212→    console.log(`[expand] ${sectionName}: final ${finalWords} words — TARGET MET ✓`);
   213→  }
   214→
   215→  return content;
   216→}
