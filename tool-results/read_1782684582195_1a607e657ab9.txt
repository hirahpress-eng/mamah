   293→// ─── Markdown Preview with Visual Placeholders ──────────────────────────────────
   294→
   295→function MarkdownPreview({ content, stageId, onGenerateVisual }: { content: string; stageId: StageId; onGenerateVisual: (placeholderId: string) => void }) {
   296→  const { sectionStages } = useArticleStore();
   297→  const stage = sectionStages.find((s) => s.id === stageId);
   298→  const placeholders = stage?.visualPlaceholders || [];
   299→
   300→  // Split content by placeholder pattern and interleave with cards
   301→  const parts: React.ReactNode[] = [];
   302→  let lastIndex = 0;
   303→  const regex = new RegExp(PLACEHOLDER_REGEX.source, 'gi');
   304→  let match;
   305→
   306→  while ((match = regex.exec(content)) !== null) {
   307→    // Text before placeholder
   308→    if (match.index > lastIndex) {
   309→      parts.push(<div key={`md-${lastIndex}`} className="prose prose-sm dark:prose-invert max-w-none"><ReactMarkdown>{content.slice(lastIndex, match.index)}</ReactMarkdown></div>);
   310→    }
   311→    // Find matching placeholder
   312→    const phType = match[1].toLowerCase() as 'figure' | 'table';
   313→    const phDesc = match[2].trim();
   314→    const ph = placeholders.find((p) => p.description === phDesc && p.type === phType);
   315→    if (ph) {
   316→      parts.push(<VisualPlaceholderCard key={ph.id} placeholder={ph} onGenerate={onGenerateVisual} stageId={stageId} />);
   317→    } else {
   318→      parts.push(
   319→        <div key={`placeholder-${match.index}`} className="my-3 p-3 rounded-lg border border-dashed border-muted-foreground/30 text-sm text-muted-foreground">
   320→          <span className="capitalize">{phType}:</span> {phDesc}
   321→        </div>
   322→      );
   323→    }
   324→    lastIndex = match.index + match[0].length;
   325→  }
   326→
   327→  // Remaining text
   328→  if (lastIndex < content.length) {
   329→    parts.push(<div key={`md-end`} className="prose prose-sm dark:prose-invert max-w-none"><ReactMarkdown>{content.slice(lastIndex)}</ReactMarkdown></div>);
   330→  }
   331→
   332→  return <div className="space-y-1">{parts}</div>;
   333→}

   435→      <motion.div key={`${stage.id}-error`} initial={{ opacity: 0, y: 12 }} animate={{ opacity: 1, y: 0 }} exit={{ opacity: 0, y: -12 }} className="flex flex-col items-center justify-center py-16 text-center">

   469→        </Card>
   470→      </ScrollArea>
   471→      {!isLastStage && (
   472→        <div className="flex justify-end pt-2">
   473→          <Button onClick={onNext} className="bg-gradient-to-r from-emerald-500 to-teal-500 hover:from-emerald-600 hover:to-teal-600 text-white gap-2">

   475→          </Button>

   543→export default function Step3Method() {
   544→  const {
   545→    references, selectedTitle, selectedKeywords, researchMethod, additionalInstructions,
   546→    sectionStages, currentStageIndex, setCurrentStageIndex, updateStage, resetStages,
   547→    compileArticleFromStages, nextStep,
   548→  } = useArticleStore();
   549→
   550→  const [view, setView] = useState<ViewMode>('config');
   551→  const [selectedEngine, setSelectedEngine] = useState<AIEngineId>('zai');
   552→  const [statusText, setStatusText] = useState('');
   553→  const abortRef = useRef<AbortController | null>(null);
   554→  const visualAbortRef = useRef<AbortController | null>(null);
   555→
   556→  const selectedRefs = useMemo(() => references.filter((r) => r.isSelected), [references]);
   557→  const canGenerate = selectedRefs.length >= 5;
   558→  const currentStage = sectionStages[currentStageIndex];
   559→  const allDone = sectionStages.every((s) => s.status === 'done');
   560→  const completedStages = sectionStages.filter((s) => s.status === 'done' && s.id !== currentStage?.id);
   561→  const isLastStage = currentStageIndex === sectionStages.length - 1;
   562→
   563→  // Switch to compile view when all done
   564→  React.useEffect(() => {
   565→    if (allDone && view === 'generating') {
   566→      setView('compile');
   567→    }
   568→  }, [allDone, view]);
   569→
   570→  // Auto-advance to next pending stage after completion
   571→  React.useEffect(() => {
   572→    if (currentStage?.status === 'done' && !allDone) {
   573→      const nextIdx = sectionStages.findIndex((s, i) => i > currentStageIndex && s.status === 'pending');
   574→      if (nextIdx !== -1) {
   575→        setCurrentStageIndex(nextIdx);
   576→      } else {
   577→        // Check if any stage before current is pending (regenerate scenario)
   578→        const prevPending = sectionStages.findIndex((s, i) => i < currentStageIndex && s.status === 'pending');
   579→        if (prevPending === -1) {
   580→          // All done or all handled
   581→        }
   582→      }
   583→    }
   584→  }, [currentStage?.status, allDone, currentStageIndex, sectionStages, setCurrentStageIndex]);
   585→
   586→  const handleStartGeneration = useCallback(() => {
   587→    if (!canGenerate) {
   588→      toast.error('Please select at least 5 references.');
   589→      return;
   590→    }
   591→    resetStages();
   592→    setCurrentStageIndex(0);
   593→    setView('generating');
   594→    toast.success('Generation started! Generate each section step by step.');
   595→  }, [canGenerate, resetStages, setCurrentStageIndex]);
   596→
   597→  const handleGenerateStage = useCallback(async () => {
   598→    if (!currentStage || currentStage.status === 'generating') return;
   599→
   600→    const abort = new AbortController();
   601→    abortRef.current = abort;
   602→
   603→    updateStage(currentStage.id, {
   604→      status: 'generating',
   605→      error: undefined,
   606→      content: '',
   607→      wordCount: 0,
   608→      visualPlaceholders: [],
   609→      startedAt: new Date().toISOString(),
   610→    });
   611→    setStatusText(`Preparing to generate ${currentStage.label}...`);
   612→
   613→    try {
   614→      // 1. POST to create a generation job (returns immediately)
   615→      const postRes = await fetch('/api/article/generate-section', {
   616→        method: 'POST',
   617→        headers: { 'Content-Type': 'application/json' },
   618→        body: JSON.stringify({
   619→          stageId: currentStage.id,
   620→          title: selectedTitle,
   621→          keywords: selectedKeywords,
   622→          references: selectedRefs,
   623→          researchMethod,
   624→          additionalInstructions,
   625→          engineId: selectedEngine,
   626→          previousSections: completedStages.map((s) => ({ id: s.id, label: s.label, content: s.content })),
   627→        }),
   628→        signal: abort.signal,
   629→      });
   630→
   631→      const postData = await postRes.json();
   632→      if (!postData.success || !postData.jobId) {
   633→        throw new Error(postData.error || 'Failed to start generation job');
   634→      }
   635→
   636→      const jobId = postData.jobId;
   637→      setStatusText(`Generation job started, waiting for ${currentStage.label}...`);
   638→
   639→      // 2. Poll for result every 3 seconds
   640→      const POLL_INTERVAL = 3000;
   641→      const MAX_POLL_TIME = 10 * 60 * 1000; // 10 minutes max
   642→      const startTime = Date.now();
   643→
   644→      while (Date.now() - startTime < MAX_POLL_TIME) {
   645→        if (abort.signal.aborted) throw new DOMException('Aborted', 'AbortError');
   646→        await new Promise((r) => setTimeout(r, POLL_INTERVAL));
   647→        if (abort.signal.aborted) throw new DOMException('Aborted', 'AbortError');
   648→
   649→        const pollRes = await fetch(`/api/article/generate-section?jobId=${jobId}`, {
   650→          signal: abort.signal,
   651→        });
   652→        const pollData = await pollRes.json();
   653→
   654→        if (!pollData.success) throw new Error(pollData.error || 'Polling failed');
   655→
   656→        // Update status text from server
   657→        if (pollData.statusMessage) {
   658→          setStatusText(pollData.statusMessage);
   659→        }
   660→
   661→        if (pollData.status === 'done' && pollData.result) {
   662→          updateStage(currentStage.id, {
   663→            status: 'done',
   664→            content: pollData.result.content,
   665→            wordCount: pollData.result.wordCount || 0,
   666→            visualPlaceholders: (pollData.result.visualPlaceholders || []).map((p: { id: string; type: 'figure' | 'table'; description: string }) => ({
   667→              ...p,
   668→              generated: false,
   669→            })),
   670→            completedAt: new Date().toISOString(),
   671→          });
   672→          toast.success(`${currentStage.label} generated successfully!`);
   673→          return;
   674→        }
   675→
   676→        if (pollData.status === 'error') {
   677→          throw new Error(pollData.error || 'Generation failed on server');
   678→        }
   679→      }
   680→
   681→      throw new Error('Generation timed out after 10 minutes');
   682→    } catch (err: unknown) {
   683→      if (err instanceof DOMException && err.name === 'AbortError') {
   684→        updateStage(currentStage.id, { status: 'pending', error: undefined });
   685→        toast.info(`${currentStage.label} generation cancelled.`);
   686→        return;
   687→      }
   688→      const message = err instanceof Error ? err.message : 'Unknown error';
   689→      updateStage(currentStage.id, { status: 'error', error: message });
   690→      toast.error(`Failed to generate ${currentStage.label}: ${message}`);
   691→    } finally {
   692→      abortRef.current = null;
   693→    }
   694→  }, [currentStage, selectedTitle, selectedKeywords, selectedRefs, researchMethod, additionalInstructions, selectedEngine, completedStages, updateStage]);
   695→
   696→  const handleCancelGeneration = useCallback(() => {
   697→    abortRef.current?.abort();
   698→  }, []);
   699→
   700→  const handleRegenerate = useCallback(() => {
   701→    if (!currentStage) return;
   702→    updateStage(currentStage.id, { status: 'pending', content: '', wordCount: 0, visualPlaceholders: [], error: undefined });
   703→  }, [currentStage, updateStage]);
   704→
   705→  const handleNext = useCallback(() => {
   706→    if (currentStageIndex < sectionStages.length - 1) {
   707→      setCurrentStageIndex(currentStageIndex + 1);
   708→    }
   709→  }, [currentStageIndex, sectionStages.length, setCurrentStageIndex]);
   710→
   711→  const handleStageClick = useCallback((index: number) => {
   712→    const stage = sectionStages[index];
   713→    if (stage.status === 'done' || stage.status === 'error' || stage.status === 'pending') {
   714→      setCurrentStageIndex(index);
   715→    }
   716→  }, [sectionStages, setCurrentStageIndex]);
   717→
   718→  const handleGenerateVisual = useCallback(async (placeholderId: string) => {
   719→    const stage = sectionStages.find((s) => s.id === currentStage?.id);
   720→    if (!stage) return;
   721→    const placeholder = stage.visualPlaceholders.find((p) => p.id === placeholderId);
   722→    if (!placeholder || placeholder.generated) return;
   723→
   724→    const abort = new AbortController();
   725→    visualAbortRef.current = abort;
   726→
   727→    const failPlaceholder = (errMsg: string) => {
   728→      const updatedPlaceholders = stage.visualPlaceholders.map((p) =>
   729→        p.id === placeholderId ? { ...p, error: errMsg } : p
   730→      );
   731→      updateStage(stage.id, { visualPlaceholders: updatedPlaceholders });
   732→    };
   733→
   734→    try {
   735→      // POST to start visual generation job
   736→      const postRes = await fetch('/api/article/generate-visual', {
   737→        method: 'POST',
   738→        headers: { 'Content-Type': 'application/json' },
   739→        body: JSON.stringify({
   740→          type: placeholder.type,
   741→          description: placeholder.description,
   742→          context: stage.content.substring(0, 500),
   743→          articleTitle: selectedTitle,
   744→          engineId: selectedEngine,
   745→        }),
   746→        signal: abort.signal,
   747→      });
   748→
   749→      const postData = await postRes.json();
   750→      if (!postData.success || !postData.jobId) {
   751→        failPlaceholder(postData.error || 'Failed to start visual generation');
   752→        toast.error(postData.error || 'Failed to start visual generation');
   753→        return;
   754→      }
   755→
   756→      const jobId = postData.jobId;
   757→
   758→      // Poll for result every 3 seconds
   759→      const POLL_INTERVAL = 3000;
   760→      const MAX_POLL_TIME = 5 * 60 * 1000;
   761→      const startTime = Date.now();
   762→
   763→      while (Date.now() - startTime < MAX_POLL_TIME) {
   764→        if (abort.signal.aborted) return;
   765→        await new Promise((r) => setTimeout(r, POLL_INTERVAL));
   766→        if (abort.signal.aborted) return;
   767→
   768→        const pollRes = await fetch(`/api/article/generate-visual?jobId=${jobId}`, {
   769→          signal: abort.signal,
   770→        });
   771→        const pollData = await pollRes.json();
   772→
   773→        if (!pollData.success) {
   774→          failPlaceholder(pollData.error || 'Polling failed');
   775→          toast.error(pollData.error || 'Polling failed');
   776→          return;
   777→        }
   778→
   779→        if (pollData.status === 'done' && pollData.result) {
   780→          const updatedPlaceholders = stage.visualPlaceholders.map((p) =>
   781→            p.id === placeholderId ? { ...p, generated: true, data: pollData.result.data, error: undefined } : p
   782→          );
   783→          updateStage(stage.id, { visualPlaceholders: updatedPlaceholders });
   784→          toast.success(`${placeholder.type === 'figure' ? 'Figure' : 'Table'} generated!`);
   785→          return;
   786→        }
   787→
   788→        if (pollData.status === 'error') {
   789→          failPlaceholder(pollData.error || 'Generation failed');
   790→          toast.error(pollData.error || 'Generation failed');
   791→          return;
   792→        }
   793→      }
   794→
   795→      failPlaceholder('Visual generation timed out');
   796→      toast.error('Visual generation timed out');
   797→    } catch (err: unknown) {
   798→      if (err instanceof DOMException && err.name === 'AbortError') return;
   799→      const message = err instanceof Error ? err.message : 'Unknown error';
   800→      failPlaceholder(message);
   801→      toast.error(message);
   802→    }
   803→  }, [sectionStages, currentStage, selectedTitle, selectedEngine, updateStage]);
   804→
   805→  const handleCompile = useCallback(() => {
   806→    compileArticleFromStages();
   807→    nextStep();
   808→    toast.success('Article compiled! Moving to preview.');
   809→  }, [compileArticleFromStages, nextStep]);
   810→
   811→  const handleRegenerateAll = useCallback(() => {
   812→    resetStages();
   813→    setCurrentStageIndex(0);
   814→    setView('generating');
   815→  }, [resetStages, setCurrentStageIndex]);
   816→
   817→  // ─── View 1: Configuration ──────────────────────────────────────────────────
   818→  if (view === 'config') {
   819→    return (
   820→      <div className="space-y-6">
   821→        {/* Step Progress Indicator */}
   822→        <div className="flex items-center justify-center">
   823→          <div className="flex items-center gap-0">
   824→            <div className="flex items-center justify-center w-8 h-8 rounded-full bg-emerald-500 text-white"><Check className="size-3.5" /></div>
   825→            <div className="w-12 h-0.5 bg-emerald-500" />
   826→            <div className="flex items-center justify-center w-8 h-8 rounded-full bg-emerald-500 text-white"><Check className="size-3.5" /></div>
   827→            <div className="w-12 h-0.5 bg-emerald-500" />
   828→            <div className="flex items-center justify-center w-8 h-8 rounded-full bg-emerald-500 text-white pulse-dot"><span className="text-[10px] font-bold">3</span></div>
   829→          </div>
   830→        </div>
   831→        <div className="flex items-center justify-center gap-8 text-[10px] font-medium text-muted-foreground -mt-1">
   832→          <span>Define</span><span>References</span><span className="text-emerald-600 dark:text-emerald-400">Generate</span>
   833→        </div>
   834→
   835→        <ReferencesSummary />
   836→        <Separator />
   837→        <MethodSelection />
   838→        <Separator />
   839→
   840→        {/* Engine Selector */}
