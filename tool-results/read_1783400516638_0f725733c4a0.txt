    91тЖТ    91тЖТ/** Animated counter hook using requestAnimationFrame */
    92тЖТ    92тЖТfunction useAnimatedCount(target: number, duration = 600) {
    93тЖТ    93тЖТ  const [displayCount, setDisplayCount] = useState(0);
    94тЖТ    94тЖТ  const rafRef = useRef<number>(0);
    95тЖТ    95тЖТ  const prevTargetRef = useRef(target);
    96тЖТ    96тЖТ
    97тЖТ    97тЖТ  useEffect(() => {
    98тЖТ    98тЖТ    prevTargetRef.current = target;
    99тЖТ    99тЖТ    const startTime = performance.now();
   100тЖТ   100тЖТ
   101тЖТ   101тЖТ    const animate = (now: number) => {
   102тЖТ   102тЖТ      const elapsed = now - startTime;
   103тЖТ   103тЖТ      const progress = Math.min(elapsed / duration, 1);
   104тЖТ   104тЖТ    // Cubic ease-out
   105тЖТ   105тЖТ    const eased = 1 - Math.pow(1 - progress, 3);
   106тЖТ   106тЖТ    const current = Math.round(eased * target);
   107тЖТ   107тЖТ    setDisplayCount(current);
   108тЖТ   108тЖТ    if (progress < 1) {
   109тЖТ   109тЖТ      rafRef.current = requestAnimationFrame(animate);
   110тЖТ   110тЖТ    }
   111тЖТ   111тЖТ    };
   112тЖТ   112тЖТ    rafRef.current = requestAnimationFrame(animate);
   113тЖТ   113тЖТ    return () => cancelAnimationFrame(rafRef.current);
   114тЖТ   114тЖТ  }, [target, duration]);
   115тЖТ   115тЖТ
   116тЖТ   116тЖТ  return displayCount;
   117тЖТ   117тЖТ}
