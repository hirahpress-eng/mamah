   341→function calculateChangePercentage(original: string, polished: string): number {
   342→  const normalize = (text: string): string[] =>
   343→    text
   344→      .toLowerCase()
   345→      .replace(/[^a-z0-9\s]/g, '')
   346→      .split(/\s+/)
   347→      .filter((w: string) => w.length > 0);
   348→
   349→  const originalWords = normalize(original);
   350→  const polishedWords = normalize(polished);
   351→
   352→  if (originalWords.length === 0) return 0;
   353→
   354→  // Use a set-based approach: count words in polished that are NOT in the same position in original
   355→  let changedWords = 0;
   356→
   357→  // For each word in polished, check if it differs from the corresponding word in original
   358→  const maxLen = Math.max(originalWords.length, polishedWords.length);
   359→  for (let i = 0; i < maxLen; i++) {
   360→    const oWord = i < originalWords.length ? originalWords[i] : null;
   361→    const pWord = i < polishedWords.length ? polishedWords[i] : null;
   362→    if (oWord !== pWord) {
   363→      changedWords++;
   364→    }
   365→  }
   366→
   367→  return Math.round((changedWords / maxLen) * 100);
   368→}

   446→
   447→  // Word count change
   448→  const origWords = original.split(/\s+/).filter(w => w.length > 0).length;
   449→  const polishedWords = polished.split(/\s+/).filter(w => w.length > 0).length;
   450→  const wordDiff = polishedWords - origWords;
   451→  if (Math.abs(wordDiff) > 20) {
   452→    changes.push(
   453→      wordDiff > 0
   454→        ? `expanded by ${wordDiff} words`
   455→        : `condensed by ${Math.abs(wordDiff)} words`,
   456→    );
   457→  }
   458→
   459→  // Vocabulary richness proxy: unique words ratio
   460→  const origUnique = new Set(origLower.split(/\s+/)).size;
   461→  const polishedUnique = new Set(polishedLower.split(/\s+/)).size;
   462→  if (polishedUnique > origUnique * 1.05) {
   463→    changes.push('vocabulary diversity improved');
   464→  }
   465→
   466→  return changes.join(', ');
   467→}

   473→function buildReviewerContext(reviewerNotes: any): string {
   474→  if (!reviewerNotes || !reviewerNotes.reviewers) return '';
   475→  let ctx = '\n\n## CRITICAL PEER REVIEWER FEEDBACK (must address ALL of these)\n\n';
   476→  ctx += `**Overall Recommendation: ${reviewerNotes.overallRecommendation}** (Score: ${reviewerNotes.overallScore}/10)\n\n`;
   477→  if (reviewerNotes.priorityActions?.length > 0) {
   478→    ctx += '**PRIORITY ACTIONS:**\n';
   479→    reviewerNotes.priorityActions.forEach((a: string, i: number) => { ctx += `${i + 1}. ${a}\n`; });
   480→    ctx += '\n';
   481→  }
   482→  reviewerNotes.reviewers.forEach((r: any) => {
   483→    ctx += `### ${r.name} (${r.role}) — ${r.recommendation}\n`;
   484→    ctx += `Assessment: ${r.overallAssessment}\n`;
   485→    if (r.majorIssues?.length > 0) {
   486→      ctx += 'Major issues to fix:\n';
   487→      r.majorIssues.forEach((issue: string, i: number) => { ctx += `  ${i + 1}. ${issue}\n`; });
   488→    }
   489→    if (r.minorIssues?.length > 0) {
   490→      ctx += 'Minor issues to fix:\n';
   491→      r.minorIssues.forEach((issue: string, i: number) => { ctx += `  ${i + 1}. ${issue}\n`; });
   492→    }
   493→    ctx += '\n';
   494→  });
   495→  return ctx;
   496→}

   498→function runPolishJob(
   499→  jobId: string,
   500→  article: any,
   501→  options: PolishOptions,
   502→  engineId: AIEngineId,
   503→  reviewerNotes?: any,
   504→) {
   505→  const job = jobs.get(jobId)!;
   506→
   507→  (async () => {
   508→    try {
   509→      const polishOptions: PolishOptions = options || {
   510→        structural: true,
   511→        tone: true,
   512→        citations: true,
   513→        coherence: true,
   514→        clarity: true,
   515→        vocabulary: true,
   516→        grammar: true,
   517→        formatting: true,
   518→      };
   519→
   520→      // Use zai as default polish engine for reliability, fall back to gemini, then grok
   521→      const polishEngine: AIEngineId = engineId || 'zai';
   522→      const validEngine: AIEngineId = AI_ENGINES.some(e => e.id === polishEngine)
   523→        ? polishEngine
   524→        : 'zai';
   525→
   526→      const keywords: string[] = article.keywords || [];
   527→
   528→      // Track metrics
   529→      let totalChangePercentage = 0;
   530→      let sectionsActuallyModified = 0;
   531→      const improvementsLog: string[] = [];
   532→      let sectionsFailed = 0;
   533→
   534→      // Polish each section SEQUENTIALLY with delays to avoid rate limit exhaustion
   535→      const sectionsToPolish = article.sections.filter(
   536→        (s: PolishSection) => s.type !== 'bibliography' && s.content?.trim().length > 0,
   537→      );
   538→
   539→      const polishedSectionsMap: Record<string, PolishSection> = {};
   540→
   541→      for (let i = 0; i < sectionsToPolish.length; i++) {
   542→        const section = sectionsToPolish[i];
   543→        const sectionName = SECTION_NAMES[section.type] || section.type;
   544→
   545→        try {
   546→          job.statusMessage = `Polishing section ${i + 1}/${sectionsToPolish.length}: ${sectionName}...`;
   547→
   548→          console.log(`[polish] Processing section ${i + 1}/${sectionsToPolish.length}: ${sectionName} (${section.wordCount || '?'} words)`);
   549→
   550→          const userPrompt = getSectionPrompt(
   551→            section.type,
   552→            section.content,
   553→            article.title,
   554→            keywords,
   555→            polishOptions,
   556→          ) + buildReviewerContext(reviewerNotes);
   557→
   558→          const maxTokens = SECTION_MAX_TOKENS[section.type] || 16000;
   559→
   560→          // First attempt
   561→          let polishedContent = await generateWithEngine(
   562→            validEngine,
   563→            BASE_SYSTEM_PROMPT,
   564→            userPrompt,
   565→            { temperature: POLISH_TEMPERATURE, maxTokens },
   566→          );
   567→
   568→          // If content is empty or generation failed, return original
   569→          if (!polishedContent || polishedContent.trim().length === 0) {
   570→            console.warn(`[polish] Section "${sectionName}" returned empty, keeping original`);
   571→            polishedSectionsMap[section.type] = section;
   572→            sectionsFailed++;
   573→            continue;
   574→          }
   575→
   576→          // Calculate change percentage
   577→          let changePct = calculateChangePercentage(section.content, polishedContent);
   578→
   579→          // If too similar (>95% identical), retry with stronger prompt
   580→          if (changePct < 5) {
   581→            console.log(
   582→              `[polish] Section "${sectionName}" only ${changePct}% changed, retrying with stronger prompt...`,
   583→            );
   584→
   585→            // Wait before retry to avoid rate limit
   586→            await new Promise<void>((r) => setTimeout(r, 8000));
   587→
   588→            const retryPrompt = `You are rewriting a ${sectionName} section for an academic article titled "${article.title}".
   589→
   590→The previous rewrite was REJECTED because it was too similar to the original. You must REWRITE this text more aggressively.
   591→
   592→ORIGINAL TEXT (do NOT copy this — REWRITE it):
   593→${section.content}
   594→
   595→REWRITE THE ABOVE TEXT. Make it sound like a different author wrote it while keeping the same meaning and all citations. Return ONLY the rewritten text.`;
   596→
   597→            const retryResult = await generateWithEngine(
   598→              validEngine,
   599→              RETRY_SYSTEM_PROMPT,
   600→              retryPrompt,
   601→              { temperature: 0.95, maxTokens },
   602→            );
   603→
   604→            if (retryResult && retryResult.trim().length > 0) {
   605→              const retryChangePct = calculateChangePercentage(
   606→                section.content,
   607→                retryResult,
   608→              );
   609→              if (retryChangePct > changePct) {
   610→                polishedContent = retryResult;
   611→                changePct = retryChangePct;
   612→                console.log(
   613→                  `[polish] Retry for "${sectionName}" achieved ${changePct}% change`,
   614→                );
   615→              }
   616→            }
   617→          }
   618→
   619→          // Track metrics
   620→          totalChangePercentage += changePct;
   621→
   622→          if (changePct >= 5) {
   623→            sectionsActuallyModified++;
   624→          }
   625→
   626→          // Generate a brief summary of what likely changed
   627→          const improvement = generateImprovementSummary(
   628→            section.type,
   629→            changePct,
   630→            section.content,
   631→            polishedContent,
   632→          );
   633→          improvementsLog.push(`${sectionName}: ${improvement}`);
   634→
   635→          const wordCount = polishedContent
   636→            .split(/\s+/)
   637→            .filter((w: string) => w.length > 0).length;
   638→
   639→          polishedSectionsMap[section.type] = {
   640→            type: section.type,
   641→            content: polishedContent,
   642→            wordCount,
   643→          };
   644→
   645→          console.log(`[polish] Section "${sectionName}" polished successfully (${changePct}% change, ${wordCount} words)`);
   646→        } catch (sectionError) {
   647→          console.error(`[polish] Error polishing "${sectionName}":`, sectionError);
   648→          // Return original section on error — don't fail the entire polish
   649→          polishedSectionsMap[section.type] = section;
   650→          sectionsFailed++;
   651→          improvementsLog.push(`${sectionName}: ERROR - kept original`);
   652→        }
   653→
   654→        // Delay between sections to avoid rate limiting (skip after the last section)
   655→        if (i < sectionsToPolish.length - 1) {
   656→          job.statusMessage = `Waiting before next section... (${i + 1}/${sectionsToPolish.length} done)`;
   657→          console.log(`[polish] Waiting ${INTER_SECTION_DELAY_MS / 1000}s before next section...`);
   658→          await new Promise<void>((r) => setTimeout(r, INTER_SECTION_DELAY_MS));
   659→        }
   660→      }
   661→
   662→      // Rebuild sections array preserving original order, including bibliography
   663→      const polishedSections: PolishSection[] = article.sections.map((section: PolishSection) => {
   664→        if (section.type === 'bibliography') return section; // Keep bibliography as-is
   665→        return polishedSectionsMap[section.type] || section; // Return polished if available, else original
   666→      });
   667→
   668→      const totalWordCount = polishedSections.reduce(
   669→        (sum: number, s: PolishSection) => sum + (s.wordCount || 0),
   670→        0,
   671→      );
   672→
   673→      // Calculate average change percentage across polished sections
   674→      const polishedSectionCount = sectionsToPolish.length;
   675→      const avgChangePercentage =
   676→        polishedSectionCount > 0
   677→          ? Math.round(totalChangePercentage / polishedSectionCount)
   678→          : 0;
   679→
   680→      const polishedArticle = {
   681→        ...article,
   682→        sections: polishedSections,
   683→        totalWordCount,
   684→        isPolished: true,
   685→      };
   686→
   687→      const changesSummary = {
   688→        structural: polishOptions.structural ?? false,
   689→        tone: polishOptions.tone ?? false,
   690→        citations: polishOptions.citations ?? false,
   691→        coherence: polishOptions.coherence ?? false,
   692→        clarity: polishOptions.clarity ?? false,
   693→        vocabulary: polishOptions.vocabulary ?? false,
   694→        grammar: polishOptions.grammar ?? false,
   695→        formatting: polishOptions.formatting ?? false,
   696→        originalWordCount: article.totalWordCount || 0,
   697→        polishedWordCount: totalWordCount,
   698→        wordCountChange: totalWordCount - (article.totalWordCount || 0),
   699→        sectionsPolished: sectionsActuallyModified,
   700→        sectionsFailed,
   701→        changePercentage: avgChangePercentage,
   702→        improvementsMade: improvementsLog.join('; '),
   703→      };
   704→
   705→      job.status = 'done';
   706→      job.statusMessage = 'Polish complete';
   707→      job.result = {
   708→        article: polishedArticle,
   709→        changes: changesSummary,
   710→      };
   711→    } catch (error: any) {
   712→      console.error(`[polish] Job ${jobId} error:`, error);
   713→      job.status = 'error';
   714→      job.error = error?.message || 'Failed to polish article';
   715→    }
   716→  })();
   717→}

   723→export async function POST(request: Request) {
   724→  try {
   725→    const body = await request.json();
   726→    const { article, options, engineId, reviewerNotes } = body;
   727→
   728→    if (!article || !article.sections || !article.title) {
   729→      return Response.json(
   730→        { success: false, error: 'Article data is required' },
   731→        { status: 400 },
   732→      );
   733→    }
   734→
   735→    const jobId = `polish_${Date.now()}_${Math.random().toString(36).slice(2, 8)}`;
   736→
   737→    const job: PolishJob = {
   738→      id: jobId,
   739→      status: 'running',
   740→      statusMessage: 'Starting polish job...',
   741→      createdAt: Date.now(),
   742→    };
   743→    jobs.set(jobId, job);
   744→
   745→    console.log(`[polish] Job ${jobId} started for "${article.title}" with ${article.sections.length} sections`);
   746→
   747→    const validEngine: AIEngineId = engineId && AI_ENGINES.some(e => e.id === engineId)
   748→      ? engineId
   749→      : 'zai';
   750→
   751→    // Fire-and-forget background polish
   752→    runPolishJob(jobId, article, options, validEngine, reviewerNotes);
   753→
   754→    // Return immediately — client will poll
   755→    return Response.json({ success: true, jobId });
   756→  } catch (error: any) {
   757→    console.error('[polish] POST error:', error);
   758→    return Response.json(
   759→      { success: false, error: error?.message || 'Internal server error' },
   760→      { status: 500 },
   761→    );
   762→  }
   763→}

   772→
   773→  if (!jobId) {
   774→    return Response.json(
   775→      { success: false, error: 'Missing jobId query parameter' },
   776→      { status: 400 },
   777→    );
   778→  }
   779→
   780→  const job = jobs.get(jobId);
   781→  if (!job) {
   782→    return Response.json(
   783→      { success: false, error: 'Job not found or expired' },
   784→      { status: 404 },
   785→    );
   786→  }
   787→
   788→  return Response.json({
   789→    success: true,
   790→    jobId: job.id,
   791→    status: job.status,
   792→    statusMessage: job.statusMessage,
   793→    result: job.result,
   794→    error: job.error,
   795→  });
