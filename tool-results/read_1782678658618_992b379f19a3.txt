   668→function GenerationProgress({ onStarted, onComplete }: { onStarted: () => void; onComplete?: () => void }) {
   669→  const {
   670→    generationProgress,
   671→    generationStatus,
   672→    isGeneratingArticle,
   673→  } = useArticleStore();
   674→
   675→  const onStartedRef = useRef(onStarted);
   676→  onStartedRef.current = onStarted;
   677→  const onCompleteRef = useRef(onComplete);
   678→  onCompleteRef.current = onComplete;
   679→
   680→  const startTimeRef = useRef<number>(Date.now());
   681→  const [, setTick] = useState(0);
   682→  const timerRef = useRef<ReturnType<typeof setInterval> | null>(null);
   683→  const [completedDuration, setCompletedDuration] = useState<number | null>(null);
   684→
   685→  // Celebration state
   686→  const celebrationTriggeredRef = useRef(false);
   687→  const [celebratedSections, setCelebratedSections] = useState<Set<number>>(new Set());
   688→  const [showCompleteBanner, setShowCompleteBanner] = useState(false);
   689→
   690→  // Start timer when generation begins
   691→  useEffect(() => {
   692→    if (isGeneratingArticle) {
   693→      startTimeRef.current = Date.now();
   694→      celebrationTriggeredRef.current = false;
   695→      requestAnimationFrame(() => {
   696→        setCelebratedSections(new Set());
   697→        setShowCompleteBanner(false);
   698→      });
   699→      timerRef.current = setInterval(() => {
   700→        setTick((prev) => prev + 1);
   701→      }, 1000);
   702→      // Trigger generation callback
   703→      onStartedRef.current();
   704→    } else {
   705→      if (timerRef.current) {
   706→        clearInterval(timerRef.current);
   707→        timerRef.current = null;
   708→        // Record final duration
   709→        const elapsed = Math.floor((Date.now() - startTimeRef.current) / 1000);
   710→        requestAnimationFrame(() => setCompletedDuration(elapsed));
   711→      }
   712→    }
   713→    return () => {
   714→      if (timerRef.current) clearInterval(timerRef.current);
   715→    };
   716→  }, [isGeneratingArticle]);
   717→
   718→  // Trigger celebration when generation completes
   719→  useEffect(() => {
   720→    if (!isGeneratingArticle && completedDuration !== null && !celebrationTriggeredRef.current) {
   721→      celebrationTriggeredRef.current = true;
   722→      // Cascading celebration: fill each section bar with 100ms delay
   723→      SECTION_ITEMS.forEach((_, idx) => {
   724→        setTimeout(() => {
   725→          requestAnimationFrame(() => {
   726→            setCelebratedSections(prev => {
   727→              const next = new Set(prev);
   728→              next.add(idx);
   729→              return next;
   730→            });
   731→          });
   732→        }, idx * 100);
   733→      });
   734→      // Show banner after cascade completes
   735→      const bannerDelay = SECTION_ITEMS.length * 100 + 300;
   736→      setTimeout(() => {
   737→        requestAnimationFrame(() => setShowCompleteBanner(true));
   738→      }, bannerDelay);
   739→      // Auto-navigate after celebration
   740→      setTimeout(() => {
   741→        onCompleteRef.current?.();
   742→      }, bannerDelay + 2000);
   743→    }
   744→  }, [isGeneratingArticle, completedDuration]);
   745→
   746→  // Compute section progress
   747→  const getSectionProgress = useCallback((item: SectionItem) => {
   748→    const idx = SECTION_ITEMS.indexOf(item);
   749→    if (celebratedSections.has(idx)) return 100;
   750→    if (generationProgress <= item.startPct) return 0;
   751→    if (generationProgress >= item.endPct) return 100;
   752→    return ((generationProgress - item.startPct) / (item.endPct - item.startPct)) * 100;
   753→  }, [generationProgress, celebratedSections]);
   754→
   755→  const getSectionStatus = useCallback((item: SectionItem): 'completed' | 'active' | 'pending' => {
   756→    const idx = SECTION_ITEMS.indexOf(item);
   757→    if (celebratedSections.has(idx) || generationProgress >= item.endPct) return 'completed';
   758→    if (generationProgress > item.startPct) return 'active';
   759→    return 'pending';
   760→  }, [generationProgress, celebratedSections]);
   761→
   762→  // Compute elapsed time from ref
   763→  const elapsedTime = Math.floor((Date.now() - startTimeRef.current) / 1000);
   764→
   765→  // Estimate time remaining
   766→  const elapsedDur = (generationProgress / 100) * TOTAL_STEP_DURATION;
   767→  const remainingDur = Math.max(0, TOTAL_STEP_DURATION - elapsedDur);
   768→  const estimatedRemainingSeconds = Math.ceil((remainingDur / TOTAL_STEP_DURATION) * 90);
   769→
   770→  const formatTime = (seconds: number) => {
   771→    const m = Math.floor(seconds / 60);
   772→    const s = seconds % 60;
   773→    return `${m}:${s.toString().padStart(2, '0')}`;
   774→  };
   775→
   776→  const isComplete = !isGeneratingArticle && completedDuration !== null;
   777→
   778→  return (
   779→    <motion.div
   780→      initial={{ opacity: 0, y: 20 }}
   781→      animate={{ opacity: 1, y: 0 }}
   782→      transition={{ duration: 0.4, ease: 'easeOut' }}
   783→      className="space-y-5 relative"
   784→    >
   785→      {/* Generation Complete Banner */}
   786→      <AnimatePresence>
   787→        {showCompleteBanner && (
   788→          <motion.div

   796→              <Sparkles className="w-5 h-5" />
