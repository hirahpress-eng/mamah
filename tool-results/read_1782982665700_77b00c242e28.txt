     1→/**
     2→ * Writing Flows — Complete cicil (incremental) generation configurations
     3→ *
     4→ * Defines the structure for each writing mode:
     5→ * - Skripsi (S1): BAB 1-5 + Referensi, 5-7 sub-steps per BAB
     6→ * - Tesis (S2): BAB 1-5 + Referensi, 5-8 sub-steps per BAB
     7→ * - Disertasi (S3): BAB 1-5 + Referensi, 8-11 sub-steps per BAB
     8→ * - Buku Ilmiah Indonesia: Book chapters
     9→ * - Buku Ilmiah English: Book chapters (English)
    10→ * - Buku Bahasa Arab: Arabic book structure
    11→ * - Buku Eksakta/Matematika: Science/Math book structure
    12→ * - Buku Keislaman: Islamic Studies book structure
    13→ * - Proposal Penelitian: Research proposal stages
    14→ * - Esai Beasiswa: Scholarship essay stages
    15→ * - Makalah: Paper stages
    16→ */
    17→
    18→// ─── Types ──────────────────────────────────────────────────────────────────
    19→
    20→export type CicilWritingMode =
    21→  | 'skripsi'
    22→  | 'tesis'
    23→  | 'disertasi'
    24→  | 'buku-id'
    25→  | 'buku-en'
    26→  | 'buku-arab'
    27→  | 'buku-eksakta'
    28→  | 'buku-keislaman'
    29→  | 'proposal'
    30→  | 'scholarship'
    31→  | 'paper';
    32→
    33→export interface CicilSubStep {
    34→  id: string;
    35→  label: string;
    36→  labelId: string;
    37→  targetWords: number;
    38→  promptFocus: string;
    39→}
    40→
    41→export interface CicilChapter {
    42→  id: string;
    43→  label: string;
    44→  labelId: string;
    45→  subSteps: CicilSubStep[];
    46→}

  1452→/** Get total number of sub-steps across all chapters */
  1453→export function getTotalSteps(mode: CicilWritingMode): number {
  1454→  return WRITING_FLOWS[mode].chapters.reduce(
  1455→    (sum, ch) => sum + ch.subSteps.length,
  1456→    0,
  1457→  );
  1458→}
  1459→
  1460→/** Get flat list of all sub-steps with chapter info */
  1461→export function getFlatSteps(mode: CicilWritingMode): Array<{
  1462→  chapterId: string;
  1463→  chapterLabel: string;
  1464→  step: CicilSubStep;
  1465→  index: number;
  1466→}> {
  1467→  const steps: Array<{
  1468→    chapterId: string;
  1469→    chapterLabel: string;
  1470→    step: CicilSubStep;
  1471→    index: number;
  1472→  }> = [];
  1473→  let idx = 0;
  1474→  for (const ch of WRITING_FLOWS[mode].chapters) {
  1475→    for (const ss of ch.subSteps) {
  1476→      steps.push({ chapterId: ch.id, chapterLabel: ch.label, step: ss, index: idx });
  1477→      idx++;
  1478→    }
  1479→  }
  1480→  return steps;
  1481→}

  1483→/** Get total target word count */
  1484→export function getTotalTargetWords(mode: CicilWritingMode): number {

  1486→    (sum, ch) => sum + ch.subSteps.reduce((s, ss) => s + ss.targetWords, 0),
  1487→    0,
  1488→  );
